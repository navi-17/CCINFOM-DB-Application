-- CORE RECORDS --
CREATE TABLE patient(
	patient_id INT AUTO_INCREMENT PRIMARY KEY,
    p_lastname VARCHAR(30) NOT NULL, 
    p_firstname VARCHAR(30) NOT NULL,
    contact_no VARCHAR(15) 
);

CREATE TABLE nurse(
	nurse_id INT AUTO_INCREMENT PRIMARY KEY,
    n_firstname VARCHAR(30) NOT NULL,
    n_lastname VARCHAR(30) NOT NULL,
	contact_no VARCHAR(15) 
);

CREATE TABLE nurse_shift(
	nurse_id INT PRIMARY KEY,
    shift_day VARCHAR(10) NOT NULL,
    start_time TIME,
    end_time TIME,
    PRIMARY KEY (nurse_id, shift_day), 
    FOREIGN KEY (nurse_id) REFERENCES nurse(nurse_id)
);

CREATE TABLE physician(
	physician_id INT AUTO_INCREMENT PRIMARY KEY,
    ph_firstname VARCHAR(30) NOT NULL,
    ph_lastname VARCHAR(30) NOT NULL,
	contact_no VARCHAR(15),
    specialization VARCHAR(20)
);

CREATE TABLE physician_schedule(
	physician_id INT PRIMARY KEY,
    schedul_day VARCHAR(10) NOT NULL,
    start_time TIME,
    end_time TIME,
    PRIMARY KEY (physician_id, schedule_day), 
    FOREIGN KEY (physician_id) REFERENCES physician(physician_id)
);

CREATE TABLE ward(
	ward_id INT AUTO_INCREMENT PRIMARY KEY, 
    floor VARCHAR(10),
    capacity INT #CLARIFY W/ GROUP
);

CREATE TABLE illness(
	illness_id INT AUTO_INCREMENT PRIMARY KEY,
    ilness_name VARCHAR(30),
    category VARCHAR(25),
    illness_description VARCHAR(50)
);

CREATE TABLE medicine(
	medicine_id INT AUTO_INCREMENT PRIMARY KEY,
    medicine_name VARCHAR(30),
    stock_qty INT
);

-- TRANSACTIONS --
CREATE TABLE diagnosis(
	diagnosis_id INT AUTO_INCREMENT PRIMARY KEY,
    patient_id INT NOT NULL,
    physician_id INT NOT NULL,
    illness_id INT NOT NULL, 
    diagnosis_date DATE, 
    notes VARCHAR (50),
    FOREIGN KEY (patient_id) REFERENCES patient(patient_id) ON DELETE RESTRICT ON UPDATE CASCADE, 
    FOREIGN KEY (physician_id) REFERENCES physician(physician_id) ON DELETE RESTRICT ON UPDATE CASCADE,
    FOREIGN KEY (illness_id) REFERENCES illness(illness_id) ON DELETE RESTRICT ON UPDATE CASCADE
);

CREATE TABLE nurse_assignment(
	nurse_assignment_id INT AUTO_INCREMENT PRIMARY KEY,
    nurse_id INT NOT NULL, 
    patient_id INT NOT NULL,
    date_assigned DATE,
    assigned_until DATE,
	FOREIGN KEY (nurse_id) REFERENCES nurse(nurse_id) ON DELETE RESTRICT ON UPDATE CASCADE,
    FOREIGN KEY (patient_id) REFERENCES patient(patient_id) ON DELETE RESTRICT ON UPDATE CASCADE,
    UNIQUE (nurse_id, patient_id, date_assigned) #You cannot assign the same nurse to the same patient more than once on the same day.
);

CREATE TABLE admission(
	admission_id INT AUTO_INCREMENT PRIMARY KEY,
    patient_id INT NOT NULL,
    ward_id INT NOT NULL,
    admission_date DATE,
    FOREIGN KEY (patient_id) REFERENCES patient(patient_id) ON DELETE RESTRICT ON UPDATE CASCADE,
    FOREIGN KEY (ward_id) REFERENCES ward(ward_id) ON DELETE RESTRICT ON UPDATE CASCADE
);

CREATE TABLE treatment(
	treatment_id INT AUTO_INCREMENT PRIMARY KEY,
    nurseAssignment_id INT NOT NULL,
    diagnosis_id INT NOT NULL,
    medicine_id INT NOT NULL,
    treatment_date DATE,
    treatment_procedure VARCHAR(30),
    remarks VARCHAR(50),
    FOREIGN KEY (nurseAssignment_id) REFERENCES nurse_assignment(nurseAssignment_id) ON DELETE RESTRICT ON UPDATE CASCADE,
    FOREIGN KEY (diagnosis_id) REFERENCES diagnosis(diagnosis_id) ON DELETE RESTRICT ON UPDATE CASCADE,
    FOREIGN KEY (medicine_id) REFERENCES medicine(medicine_id) ON DELETE SET NULL ON UPDATE CASCADE #its okay if medicine gets deleted since a patient can be treated with procedure
);

CREATE TABLE discharge(
	discharge_id INT AUTO_INCREMENT PRIMARY KEY,
    admission_id INT NOT NULL,
    discharge_date DATE,
    FOREIGN KEY (admission_id) REFERENCES admission(admission_id) ON DELETE CASCADE ON UPDATE CASCADE #if an admission is deleted, discharge also gets deleted
);
